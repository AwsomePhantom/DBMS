
MYSQL Commands
--------------------------
List tables in a database:
select table_schema, table_name, table_rows from information_schema.tables where table_schema = 'classicmodels';

List the column in a particular table:
select * from information_schema.columns where table_schema = 'classicmodels' and table_name = 'customers'

Fiter column for datatypes:
 select column_name, data_type, character_maximum_length from information_schema.columns where table_schema = 'classicmodels' and table_name = 'customers';
 
Using column alias for views:
When using multiple aliases, use apostrophe for strings
  select column_name as 'Columns', data_type as 'Type', character_maximum_length as 'Max' from information_schema.columns where table_schema = 'classicmodels' and table_name = 'customers';
  
  
Create Database
----------------------------------------------
Create DB			->	CREATE DATABASE dbname;
Drop DB				->	DROP DATABASE dbname;
Drop DB				->	DROP IF EXISTS dbname;
Create table-		->	CREATE TABLE table_name {
							column_name datatype constraint,
							column_name datatype constraint,
							column_name datatype constraint,
						};
				
Create table using another table:
	CREATE TABLE new_table_name AS
	SELECT column1, column2 FROM existing_table WHERE...;


Drop Table			->	DROP table_name;
Clear Table			-> TRUNCATE TABLE table_name;
Add Column			-> ALTER TABLE table_name ADD column_name datatype;
Drop Column			-> ALTER TABLE table_name DROP COLUMN column_name;
Modify Column		-> ALTER TABLE table_name MODIFY COLUMN column_name datatype;
Default constraint	-> CREATE TABLE table_name {
							column_name datatype DEFAULT = 'default_string' };
					   CREATE TABLE table_name {
							column_name date DEFAULT = CURRENT_DATE() };
Add a default const	-> ALTER TABLE table_name ALTER table_name SET DEFAULT 'default_string';
Drop default const	-> ALTER TABLE table_name ALTER table_name DROP DEFAULT;

Insert value		-> INSERT INTO table_name (column1, column2, column3)
					   VALUES ('value1', 'value2', 'value3');
					   INSERT INTO table_name VALUES ('value1', 'value2');
					   
Add Constraint		->	ALTER TABLE records modify id INT NOT NULL AUTO_INCREMENT PRIMARY KEY;
						ALTER TABLE `records` ADD UNIQUE(`id`);
						
Update row			->	UPDATE Customers
						SET column_name = 'value', column_name= 'value'
						WHERE column_name = 1 and column_name = 2;

Delete row			->	DELETE FROM table_name WHERE column_name='value';
						
Set autoincrement	-> ALTER TABLE table_name AUTO_INCREMENT=10

Change column order	->	ALTER TABLE table_name 
						modify column column_name datatype NOT NULL FIRST;
